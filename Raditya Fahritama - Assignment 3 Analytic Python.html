#!/usr/bin/ python3

"""this code is to create 5x5 array with and create a loop to calculate sum of diagonal of the array """

import numpy as np

"""create list and append number to the list using randn function"""
#assign empty list
randlist = []
n = 5
nn = 25

#loop and append randn numbers to the list
for i in range(nn):
    x = np.random.randn()
    randlist.append(x)

"""creating the array and reshape to 5x5"""    
#format the list to .2 
randformatted = [round(y,2) for y in randlist ]
#create and reshape the array to 5x5 
randarray = np.reshape(randformatted, (n,n))

"""this section is to sum the diagonals of the array"""
#assign sum variables
firstdiag = 0
seconddiag = 0

#loop in range of array
for i in range(0,n):
    for j in range(0,n):
        #if i equals j, add values to sum variable 
        if i == j:
            firstdiag = firstdiag + randarray[i][j]
        #if i plus j equal length of array - 1, add values to sum variable
        if i + j == n - 1:
            seconddiag = seconddiag + randarray[i][j]

"""display the results"""            
print('The random array : \n',randarray)
print('\nFirst Diagonal sum : %.2f'%firstdiag) 
print('\nSecond Diagonal sum : %.2f'%seconddiag)

#!/usr/bin/ python3

"""this code is to create random 10x10 array with integer on range 1000 and count the even numbers"""
import numpy as np

#assign size number
n = 10
#create random integer array from 0 - 1000
x = np.random.randint(0, 1000, size = (n, n)) 
print("Random integer array : \n",x,'\n')

#assign count variables
counteven = 0
countodd = 0

"""this section is to loop and see if elements are even or odd"""
for i in range(0,n):
    for j in range(0,n):
        #condition if modulo 2 is equal zero
        if (x[i][j] % 2) == 0:
            counteven += 1
        #condition if modulo 2 is not equal zero
        elif (x[i][j] % 2) != 0:
            countodd += 1
            
#display the results
print("Count of even number in the array : ",counteven)
print("Count of odd number in the array : ",countodd)

        x[i][j] = round(x[i][j], 3)

            elem = x[i][j]

#defining the function
def featureimp(columns, importances): 
    importances = pd.DataFrame(data={
        'Attribute': columns,
        'Importance': importances
    })
    importances = importances.sort_values(by='Importance', ascending=False)
    plt.figure(figsize=(5, 5))
    plt.bar(x=importances['Attribute'], height=importances['Importance'], color='salmon')
    plt.title('Features Impact Rank on MPG' , size=20)
    plt.xticks(rotation='vertical')
    plt.show()

#getting model feature importance
